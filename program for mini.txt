  const f = e.dataTransfer.files[0];
  if(f) processFile(f);
}

function processFile(file){
  if(!file.type.startsWith('image/')){
    showToast('Please upload an image file (JPG, PNG, WEBP)','err'); return;
  }
  if(ocrRunning){ showToast('OCR already running, please wait…','err'); return; }
  const reader = new FileReader();
  reader.onload = function(ev){
    const src = ev.target.result;
    document.getElementById('previewImg').src = src;
    document.getElementById('prevName').textContent = file.name;
    document.getElementById('previewBox').style.display = 'block';
    document.getElementById('uploadZone').style.display = 'none';
    document.getElementById('ocrChips').innerHTML = '';
    document.getElementById('ocrTip').style.display = 'none';
    runOCR(src);
  };
  reader.readAsDataURL(file);
}

function removeImage(){
  document.getElementById('previewBox').style.display = 'none';
  document.getElementById('uploadZone').style.display = 'block';
  document.getElementById('previewImg').src = '';
  document.getElementById('ocrChips').innerHTML = '';
  document.getElementById('ocrTip').style.display = 'none';
  document.getElementById('ocrWrap').style.display = 'none';
  cancelOCR();
}

async function runOCR(imgSrc){
  if(ocrRunning) return; // FIX: prevent double execution
  ocrRunning = true;
  ocrCancelled = false;
  const wrap = document.getElementById('ocrWrap');
  const bar  = document.getElementById('ocrBar');
  const txt  = document.getElementById('ocrTxt');
  wrap.style.display = 'block';
  bar.style.width = '0%';
  txt.textContent = 'Loading OCR engine…';
  try {
    const worker = await Tesseract.createWorker('eng', 1, {
      logger: m => {
        if(ocrCancelled) return;
        if(m.status==='loading tesseract core') txt.textContent='Loading OCR core…';
        else if(m.status==='initializing tesseract') txt.textContent='Initialising…';
        else if(m.status==='loading language traineddata') txt.textContent='Loading language data…';
        else if(m.status==='recognizing text') txt.textContent='Scanning bill… '+Math.round((m.progress||0)*100)+'%';
        if(m.progress!==undefined) bar.style.width=(m.progress*100)+'%';
      }
    });
    ocrWorker = worker;
    if(ocrCancelled){ await worker.terminate(); ocrRunning=false; return; }
    const {data} = await worker.recognize(imgSrc);
    await worker.terminate();
    ocrWorker = null;
    ocrRunning = false;
    if(ocrCancelled) return;
    wrap.style.display = 'none';
    parseOCRText(data.text);
  } catch(err){
    ocrRunning = false;
    if(!ocrCancelled){
      document.getElementById('ocrWrap').style.display = 'none';
      showToast('OCR failed: '+err.message,'err');
    }
  }
}

function cancelOCR(){
  ocrCancelled = true;
  ocrRunning = false;
  if(ocrWorker){ ocrWorker.terminate().catch(()=>{}); ocrWorker=null; }
  document.getElementById('ocrWrap').style.display = 'none';
}

function parseOCRText(text){
  const lines = text.split('\n').map(l=>l.trim()).filter(l=>l.length>2);
  const skipWords = /^(total|subtotal|tax|gst|vat|discount|amount|mrp|price|rs|inr|₹|thank|welcome|bill|invoice|receipt|date|time|cash|change|due|balance|paid|no\.|sr\.|sl\.|item|qty|unit|rate|hsn|gstin)/i;
  const pricePattern = /\d+[.,]\d{2,3}$/;
  const qtyPatterns = [
    /^(\d+(?:\.\d+)?)\s*(kg|g|gm|ml|l|ltr|litre|pcs?|nos?|pack|packet|box|bottle|bag|dozen|doz)\b/i,
    /\b(\d+(?:\.\d+)?)\s*(kg|g|gm|ml|l|ltr|litre|pcs?|nos?|pack|packet|box|bottle|bag|dozen|doz)\b/i,
    /\bx\s*(\d+)\b/i,
    /\b(\d+)\s*x\b/i,
    /^(\d+)\s+/,
  ];
  const extracted = [];
  lines.forEach(line=>{
    if(skipWords.test(line.trim())) return;
    if(pricePattern.test(line.trim())) return;
    if(/^\d+(\.\d+)?$/.test(line.trim())) return;
    let clean = line.replace(/[\d,]+\.\d{2,3}\s*$/, '').replace(/₹\s*[\d.,]+/g,'').trim();
    if(clean.length<2) return;
    let qty='', unit='';
    for(const pat of qtyPatterns){
      const m = clean.match(pat);
      if(m){ qty=m[1]; unit=m[2]||''; clean=clean.replace(m[0],'').trim(); break; }
    }
    clean = clean.replace(/[^a-zA-Z0-9\s\-\/&().]/g,' ').replace(/\s+/g,' ').trim();
    if(clean.length<2) return;
    const name = clean.split(' ').map(w=>w.charAt(0).toUpperCase()+w.slice(1).toLowerCase()).join(' ');
    extracted.push({name, qty: qty?(qty+(unit?' '+unit:'')):'1'});
  });
  if(extracted.length===0){
    showToast('Could not detect items. Try a clearer image or add manually.','err');
    document.getElementById('ocrTip').style.display='block';
    return;
  }
  const chipBox = document.getElementById('ocrChips');
  chipBox.innerHTML = extracted.map(e=>`<span class="ocr-chip">✓ ${e.name}</span>`).join('');
  document.getElementById('ocrTip').style.display = 'block';
  extracted.forEach(item => manualRows.push({name:item.name, qty:item.qty, expiry:''}));
  renderManualTable();
  showToast(`✅ Detected ${extracted.length} item(s)! Set expiry dates below.`,'ok');
}

function saveTracker(){ localStorage.setItem('trackerItems',JSON.stringify(trackerItems)); }

function daysLeft(expiry){
  if(!expiry) return null;
  const exp = new Date(expiry); exp.setHours(23,59,59);
  return Math.round((exp - new Date())/(1000*60*60*24));
}

function renderTracker(){
  const tbody = document.getElementById('trackerBody');
  if(!trackerItems.length){
    tbody.innerHTML=`<tr><td colspan="6" style="text-align:center;padding:32px;color:var(--muted);">No items yet — add from the Add Items tab</td></tr>`;
    updateStats(); return;
  }
  const sorted = [...trackerItems].sort((a,b)=>{
    if(!a.expiry) return 1; if(!b.expiry) return -1;
    return new Date(a.expiry)-new Date(b.expiry);
  });
  tbody.innerHTML = sorted.map(item=>{
    const dl = daysLeft(item.expiry);
    let dotClass='dot-none', daysText='—';
    if(dl!==null){
      if(dl<0){ dotClass='dot-danger'; daysText=`<span style="color:var(--danger);font-weight:700;">Expired ${Math.abs(dl)}d ago</span>`; }
      else if(dl<=3){ dotClass='dot-warn'; daysText=`<span style="color:var(--warn);font-weight:700;">${dl}d left</span>`; }
      else{ dotClass='dot-ok'; daysText=`<span style="color:var(--accent);">${dl}d left</span>`; }
    }
    return `<tr>
      <td><span class="dot ${dotClass}"></span></td>
      <td><input type="text" value="${escHtml(item.name)}" onchange="updateTrackerField('${item.id}','name',this.value)" style="background:transparent;border:none;padding:0;font-size:.88rem;color:var(--text);width:100%;min-width:90px;"></td>
      <td><input type="text" value="${escHtml(item.qty||'')}" onchange="updateTrackerField('${item.id}','qty',this.value)" style="background:transparent;border:none;padding:0;font-size:.88rem;color:var(--text);width:80px;"></td>
      <td><input type="date" value="${item.expiry||''}" onchange="updateTrackerField('${item.id}','expiry',this.value)"></td>
      <td>${daysText}</td>
      <td><button class="del-btn" onclick="deleteTrackerItem('${item.id}')">🗑</button></td>
    </tr>`;
  }).join('');
  updateStats();
  renderAlertBanners();
}

function updateTrackerField(id,field,val){
  const item = trackerItems.find(i=>i.id===id);
  if(item){ item[field]=val; saveTracker(); }
}
function deleteTrackerItem(id){
  trackerItems = trackerItems.filter(i=>i.id!==id);
  saveTracker(); renderTracker();
}
function addTrackerRow(){
  trackerItems.push({id:String(Date.now())+String(Math.floor(Math.random()*1e6)),name:'New Item',qty:'1',expiry:'',added:new Date().toISOString()});
  saveTracker(); renderTracker();
}
function clearExpired(){
  const before = trackerItems.length;
  trackerItems = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl===null||dl>=0; });
  saveTracker(); renderTracker();
  showToast(`Removed ${before-trackerItems.length} expired item(s)`,'ok');
}
function updateStats(){
  let fresh=0,soon=0,exp=0;
  trackerItems.forEach(i=>{ const dl=daysLeft(i.expiry); if(dl===null)return; if(dl<0)exp++; else if(dl<=3)soon++; else fresh++; });
  document.getElementById('s-total').textContent = trackerItems.length;
  document.getElementById('s-fresh').textContent = fresh;
  document.getElementById('s-soon').textContent  = soon;
  document.getElementById('s-exp').textContent   = exp;
}
function renderAlertBanners(){
  const box = document.getElementById('alertsBox');
  const expired = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl<0; });
  const soon    = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl>=0&&dl<=3; });
  let html='';
  if(expired.length) html+=`<div class="alert-banner alert-danger">🚨 <div><strong>${expired.length} item(s) expired:</strong> ${expired.map(i=>i.name).join(', ')}</div></div>`;
  if(soon.length)    html+=`<div class="alert-banner alert-warn">⚠️ <div><strong>${soon.length} item(s) expiring within 3 days:</strong> ${soon.map(i=>i.name).join(', ')}</div></div>`;
  if(!expired.length&&!soon.length&&trackerItems.length) html=`<div class="alert-banner alert-ok">✅ All items are fresh!</div>`;
  box.innerHTML = html;
}

function toggleAutoAlert(){
  autoAlertEnabled = document.getElementById('autoAlertToggle').checked;
  localStorage.setItem('autoAlert', autoAlertEnabled?'1':'0');
  if(autoAlertEnabled){
    startAlertScheduler();
    showToast('✅ Auto-alert scheduler enabled!','ok');
  } else {
    stopAlertScheduler();
    showToast('Auto-alert disabled','ok');
  }
  updateAutoAlertUI();
}

function startAlertScheduler(){
  stopAlertScheduler();
  // Check every minute
  autoAlertInterval = setInterval(checkScheduledAlert, 60000);
  checkScheduledAlert(); // check immediately on enable
}

function stopAlertScheduler(){
  if(autoAlertInterval){ clearInterval(autoAlertInterval); autoAlertInterval=null; }
}

function checkScheduledAlert(){
  if(!autoAlertEnabled) return;
  const now = new Date();
  const currentTime = now.getHours().toString().padStart(2,'0')+':'+now.getMinutes().toString().padStart(2,'0');
  const scheduledTime = document.getElementById('alertTimeInput').value || '09:00';
  const days = document.getElementById('alertDaysSelect').value;
  const dayOfWeek = now.getDay(); // 0=Sun, 6=Sat
  // Check day filter
  if(days==='weekdays' && (dayOfWeek===0||dayOfWeek===6)) return;
  if(days==='weekends' && dayOfWeek>=1 && dayOfWeek<=5) return;
  // Check time match
  if(currentTime !== scheduledTime) return;
  // Check if already fired today
  const lastFired = localStorage.getItem('lastAlertFired');
  const todayKey = now.toDateString()+scheduledTime;
  if(lastFired === todayKey) return;
  localStorage.setItem('lastAlertFired', todayKey);
  // Fire alert
  const method = document.getElementById('alertMethodSelect').value;
  fireAutoAlert(method);
}

function fireAutoAlert(method){
  const expired = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl<0; });
  const soon    = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl>=0&&dl<=3; });
  if(!expired.length && !soon.length){ addAlertLog('⏰ Scheduled check: All items fresh, no alert needed.'); return; }
  addAlertLog(`⏰ Auto-alert fired via ${method}: ${expired.length} expired, ${soon.length} expiring soon`);
  if(method==='whatsapp') sendWaAlertNow('all');
  else if(method==='sms') sendSmsNow('all');
  else if(method==='email') sendEmailNow('all');
  else if(method==='browser'){
    sendBrowserNotif('🚨 Expiry Alert!',
      (expired.length?`Expired: ${expired.map(i=>i.name).join(', ')}. `:'') +
      (soon.length?`Expiring soon: ${soon.map(i=>i.name).join(', ')}.`:'')
    );
  }
}

function updateAutoAlertUI(){
  const badge = document.getElementById('autoAlertStatusBadge');
  const nextInfo = document.getElementById('nextAlertInfo');
  if(autoAlertEnabled){
    const t = document.getElementById('alertTimeInput').value;
    const d = document.getElementById('alertDaysSelect').value;
    const m = document.getElementById('alertMethodSelect').value;
    badge.innerHTML = `<span class="auto-status-on">✅ Scheduler active — ${t} · ${d} · via ${m}</span>`;
    nextInfo.textContent = `Next check at ${t} (tab must be open)`;
  } else {
    badge.innerHTML = `<span class="auto-status-off">⏸ Scheduler off — toggle to enable</span>`;
    nextInfo.textContent = '';
  }
}

function renderAlerts(){
  const status   = document.getElementById('browserNotifStatus');
  const btnGrant = document.getElementById('btnGrantNotif');
  const btnTest  = document.getElementById('btnTestNotif');
  if(!('Notification' in window)){
    status.innerHTML=`<div style="font-size:.8rem;color:var(--danger);margin-bottom:8px;">❌ Browser does not support notifications.</div>`;
    btnGrant.style.display='none'; return;
  }
  if(Notification.permission==='granted'){
    status.innerHTML=`<div style="font-size:.8rem;color:var(--accent);margin-bottom:8px;">✅ Browser notifications <strong>enabled</strong>!</div>`;
    btnGrant.style.display='none'; btnTest.style.display='inline-flex';
  } else if(Notification.permission==='denied'){
    status.innerHTML=`<div style="font-size:.8rem;color:var(--danger);margin-bottom:8px;">🚫 Blocked — allow in browser settings.</div>`;
    btnGrant.style.display='none';
  } else {
    status.innerHTML=`<div style="font-size:.8rem;color:var(--muted);margin-bottom:8px;">🔔 Click below to allow expiry pop-up alerts.</div>`;
    btnGrant.style.display='inline-flex'; btnTest.style.display='none';
  }
  document.getElementById('autoAlertToggle').checked = autoAlertEnabled;
  updateAutoAlertUI();
  renderWaSection();
  renderSmsSection();
  renderEmailSection();
  if(autoAlertEnabled) startAlertScheduler();
}

function requestBrowserNotif(){
  Notification.requestPermission().then(p=>{
    renderAlerts();
    if(p==='granted'){ showToast('✅ Notifications enabled!','ok'); sendBrowserNotif('Smart Expiry Tracker','Ready! Expiry alerts will appear here.'); }
    else if(p==='denied') showToast('Notifications blocked','err');
  });
}
function sendTestBrowserNotif(){
  const expired = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl<0; });
  const soon    = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl>=0&&dl<=3; });
  if(!expired.length&&!soon.length) sendBrowserNotif('✅ All Fresh!','No expiry alerts needed right now.');

  }
  showToast('Test notification sent!','ok');
}

function buildAlertText(type){
  const today = new Date().toLocaleDateString('en-IN',{day:'2-digit',month:'short',year:'numeric'});
  const expired = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl<0; });
  const soon    = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl>=0&&dl<=3; });
  let msg = `🥦 Smart Expiry Tracker Alert\n📅 ${today}\n\n`;
  if(type==='expired'||type==='all'){
    if(expired.length){ msg+='🚨 EXPIRED:\n'; expired.forEach(i=>{ const dl=daysLeft(i.expiry); msg+=`• ${i.name} (${i.qty}) — ${Math.abs(dl)}d ago\n`; }); msg+='\n'; }
  }
  if(type==='soon'||type==='all'){
    if(soon.length){ msg+='⚠️ EXPIRING SOON (3 days):\n'; soon.forEach(i=>{ const dl=daysLeft(i.expiry); msg+=`• ${i.name} (${i.qty}) — ${dl===0?'today':dl+'d left'}\n`; }); msg+='\n'; }
  }
  if(type==='all'){
    const fresh = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl>3; });
    if(fresh.length){ msg+='✅ FRESH:\n'; fresh.forEach(i=>{ msg+=`• ${i.name} (${i.qty})\n`; }); }
  }
  msg += '\n_Sent from Smart Expiry Tracker_';
  return msg;
}

function sendWaAlertNow(type){
  if(!waPhone){ showToast('Save your WhatsApp number first!','err'); return; }
  const expired = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl<0; });
  const soon    = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl>=0&&dl<=3; });
  if(type==='expired'&&!expired.length){ showToast('No expired items!','ok'); return; }
  if(type==='soon'&&!soon.length){ showToast('No items expiring soon!','ok'); return; }
  if(type==='all'&&!trackerItems.length){ showToast('No items in tracker!','err'); return; }
  window.open('https://wa.me/'+waPhone+'?text='+encodeURIComponent(buildAlertText(type)),'_blank');
  addAlertLog('📲 WhatsApp alert opened: '+type);
  showToast('📲 WhatsApp opened — tap Send!','ok');
}

  const expired = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl<0; });
  const soon    = trackerItems.filter(i=>{ const dl=daysLeft(i.expiry); return dl!==null&&dl>=0&&dl<=3; });
  if(type==='expired'&&!expired.length){ showToast('No expired items!','ok'); return; }
  if(type==='soon'&&!soon.length){ showToast('No items expiring soon!','ok'); return; }
  const subject = 'Expiry Alert — Smart Tracker';
  const body    = buildAlertText(type);
  const mailto  = `mailto:${smsGateway}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;
  window.open(mailto,'_blank');
  addAlertLog('💬 SMS (email gateway) opened: '+type+' → '+smsGateway);
  showToast('📧 Email app opened — tap Send to deliver SMS!','ok');
}

function saveEmail(){
  const val = document.getElementById('emailInput').value.trim();
  if(!val||!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val)){ showToast('Enter a valid email address','err'); return; }
  alertEmail = val;
  localStorage.setItem('alertEmail', alertEmail);
  renderEmailSection();
  showToast('✅ Email saved!','ok');
  addAlertLog('📧 Alert email saved: '+alertEmail);
}

function addAlertLog(msg){
  alertLog.unshift({time:new Date().toLocaleString('en-IN'),msg});
  if(alertLog.length>30) alertLog=alertLog.slice(0,30);
  localStorage.setItem('alertLog',JSON.stringify(alertLog));
}
function renderAlertLog(){
  const box = document.getElementById('alertLog');
  if(!alertLog.length){ box.innerHTML='<div style="color:var(--muted);font-size:.83rem;text-align:center;padding:16px 0;">No alerts sent yet</div>'; return; }
  box.innerHTML = alertLog.map(l=>`<div class="log-item"><div class="log-time">${l.time}</div><div class="log-msg">${l.msg}</div></div>`).join('');
}

function escHtml(s){ return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }
function copyText(t){ navigator.clipboard.writeText(t).then(()=>showToast('Copied: '+t,'ok')); }
function showToast(msg, type='ok'){
  document.querySelectorAll('.toast').forEach(t=>t.remove());
  const t = document.createElement('div');
  t.className = 'toast '+(type==='ok'?'tok':type==='err'?'terr':'tinfo');
  t.textContent = msg;
  document.body.appendChild(t);
  setTimeout(()=>t.remove(), 3800);
}
